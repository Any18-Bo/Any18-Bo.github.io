import tkinter as tk from tkinter import messagebox class TicTacToe: def __init__(self, master): self.master = master self.master.title("Крестики-нолики") self.current_player = 'X' self.buttons = [] self.create_widgets() def create_widgets(self): # Создаем 3x3 кнопки для игрового поля for row in range(3): button_row = [] for col in range(3): btn = tk.Button(self.master, text=' ', font=('Arial', 40), width=5, height=2, command=lambda r=row, c=col: self.on_button_click(r, c)) btn.grid(row=row, column=col) button_row.append(btn) self.buttons.append(button_row) # Кнопка для сброса игры reset_button = tk.Button(self.master, text='Сбросить', font=('Arial', 14), command=self.reset_board) reset_button.grid(row=3, column=0, columnspan=3, sticky='we') def on_button_click(self, row, col): if self.buttons[row][col]['text'] == ' ': self.buttons[row][col]['text'] = self.current_player if self.check_winner(): messagebox.showinfo("Победа!", f"Игрок {self.current_player} победил!") self.disable_buttons() elif self.check_draw(): messagebox.showinfo("Ничья", "Игра окончена вничью!") self.disable_buttons() else: self.switch_player() def check_winner(self): # Проверка строк и столбцов for i in range(3): if (self.buttons[i][0]['text'] == self.buttons[i][1]['text'] == self.buttons[i][2]['text'] != ' '): return True if (self.buttons[0][i]['text'] == self.buttons[1][i]['text'] == self.buttons[2][i]['text'] != ' '): return True # Проверка диагоналей if (self.buttons[0][0]['text'] == self.buttons[1][1]['text'] == self.buttons[2][2]['text'] != ' '): return True if (self.buttons[0][2]['text'] == self.buttons[1][1]['text'] == self.buttons[2][0]['text'] != ' '): return True return False def check_draw(self): for row in range(3): for col in range(3): if self.buttons[row][col]['text'] == ' ': return False return True def disable_buttons(self): for row in range(3): for col in range(3): self.buttons[row][col]['state'] = 'disabled' def reset_board(self): for row in range(3): for col in range(3): self.buttons[row][col]['text'] = ' ' self.buttons[row][col]['state'] = 'normal' self.current_player = 'X' def switch_player(self): self.current_player = 'O' if self.current_player == 'X' else 'X' if __name__ == "__main__": root = tk.Tk() game = TicTacToe(root) root.mainloop()
